def inititalize(
  root_dir: '.',
  title: 'GeraBlog',
  name: 'GeraBlog Static Blog Generator',
  description: 'Blog Generator - my own static site generator',
  verbose: false
)
  @root_dir = root_dir
  @title = title
  @name = name
  @description = description
  @verbose = verbose

  @output_dir = File.join @root_dir, 'output'
end

def render_blog
  posts = []

  puts @root_dir, @output_dir
  Dir.mkdir(@output_dir) unless Dir.exist?(@output_dir)

  Dir[File.join @root_dir, '*'].each do |category|
    puts category if @verbose

    Dir["#{category}/*.md"]
      .sort
      .reverse
      .map { |arq| posts.push render_page(filename: arq, category: category) }
  end

  posts
end

def render_page(filename,, category))
  puts "\t#{filename}" if @verbose
  newfile = filename.gsub %r{/md$/}, 'html'

  md_content = File.read(filename)
  lines = md_content.split("\n")
  markdown = Render.new(CustomRender.new(category))

  {
    title: lines[0][2..-1],
    description: lines[2][3..-1],
    date: Date.parse(newfile.match('\A.*/(....-..-..)')[1]).rfc822,
    content: markdown.render(md_content)
  }
end
