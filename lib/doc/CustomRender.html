def block_code(code, language)
  l = language.nil? ? @lang : language
  %Q(<pre><code class="language-#{l}">\n#{html_escape(code)}\n</code></pre>)
end

def html_escape(string)
  string.gsub(
    %r{ ['&\"<>\/] },
    '&' => '&amp;',
    '<' => '&lt;',
    '>' => '&gt;',
    '"' => '&quot;',
    "'" => '&#x27;',
    '/' => '&#x2F;'
  )
end
